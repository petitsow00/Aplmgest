Sécurité RLS
1. Profils utilisateurs & Paramètres agence
-- Extension UUID CREATE EXTENSION IF NOT EXISTS "uuid-ossp"; -- Type rôle CREATE TYPE user_role AS ENUM ('administrateur','gestionnaire','comptable','employe','standard'); -- Profils (liés à auth.users Supabase) CREATE TABLE profiles ( id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE, nom_complet TEXT NOT NULL, email TEXT NOT NULL, telephone TEXT, role user_role DEFAULT 'standard', actif BOOLEAN DEFAULT TRUE, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Paramètres agence (table singleton) CREATE TABLE parametres ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), nom_agence TEXT NOT NULL DEFAULT 'APLM BUZNESS COMPANY TRAVEL SARL', ninea TEXT, registre_commerce TEXT, adresse TEXT, telephone1 TEXT, telephone2 TEXT, whatsapp TEXT, email TEXT, site_web TEXT, devise TEXT DEFAULT 'FCFA', taux_tva NUMERIC DEFAULT 18, description TEXT, logo_url TEXT, updated_at TIMESTAMPTZ DEFAULT NOW() ); INSERT INTO parametres (nom_agence) VALUES ('APLM BUZNESS COMPANY TRAVEL SARL');
2. Clients, Engagements, Dossiers & Documents
CREATE TABLE clients ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), prenom TEXT NOT NULL, nom TEXT NOT NULL, telephone TEXT NOT NULL, email TEXT, adresse TEXT, nationalite TEXT DEFAULT 'Sénégalaise', type_document TEXT DEFAULT 'Passeport', num_document TEXT, expiry_document DATE, actif BOOLEAN DEFAULT TRUE, agent_id UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE engagements ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), numero TEXT UNIQUE NOT NULL, client_id UUID NOT NULL REFERENCES clients(id), date_engagement DATE NOT NULL DEFAULT CURRENT_DATE, pays_destination TEXT NOT NULL, type_visa TEXT NOT NULL, date_voyage DATE, montant_frais NUMERIC NOT NULL DEFAULT 0, conditions_ok BOOLEAN DEFAULT FALSE, sig_client_url TEXT, sig_agent_url TEXT, statut TEXT DEFAULT 'brouillon', agent_id UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE dossiers ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), numero TEXT UNIQUE NOT NULL, client_id UUID NOT NULL REFERENCES clients(id), engagement_id UUID REFERENCES engagements(id), type_service TEXT NOT NULL, destination TEXT, date_echeance DATE, montant NUMERIC NOT NULL DEFAULT 0, statut TEXT DEFAULT 'en_cours', etape INT DEFAULT 1, etapes_total INT DEFAULT 5, notes TEXT, agent_id UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE documents_clients ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), client_id UUID NOT NULL REFERENCES clients(id), dossier_id UUID REFERENCES dossiers(id), nom_fichier TEXT NOT NULL, type_document TEXT NOT NULL, storage_path TEXT NOT NULL, taille_ko INT, date_expiration DATE, created_at TIMESTAMPTZ DEFAULT NOW() );
3. Services, Oumrah & Transferts
CREATE TABLE services ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), code TEXT UNIQUE NOT NULL, nom TEXT NOT NULL, actif BOOLEAN DEFAULT TRUE ); INSERT INTO services (code,nom) VALUES ('visa','Procédures visa'),('oumrah','Oumrah'), ('billet','Billets avion'),('hotel','Réservation hôtel'), ('assurance','Assurance voyage'),('transfert_argent','Transfert argent'), ('transfert_aibd','Transfert AIBD'),('pelerinage_fes','Pèlerinage Fès'), ('assistance','Assistance voyage'); CREATE TABLE dossiers_oumrah ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), dossier_id UUID NOT NULL REFERENCES dossiers(id), groupe TEXT, date_depart DATE, duree_jours INT DEFAULT 21, acompte NUMERIC DEFAULT 0 ); CREATE TABLE transferts ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), reference TEXT UNIQUE NOT NULL, expediteur_nom TEXT NOT NULL, expediteur_tel TEXT, beneficiaire_nom TEXT NOT NULL, pays_destination TEXT NOT NULL, montant_fcfa NUMERIC NOT NULL, devise_reception TEXT DEFAULT 'EUR', type_transfert TEXT DEFAULT 'international', statut TEXT DEFAULT 'en_attente', agent_id UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW() );
4. Facturation & Journal SYSCOHADA
CREATE TABLE factures ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), numero TEXT UNIQUE NOT NULL, client_id UUID NOT NULL REFERENCES clients(id), dossier_id UUID REFERENCES dossiers(id), type_doc TEXT DEFAULT 'facture', date_doc DATE NOT NULL DEFAULT CURRENT_DATE, montant_ht NUMERIC NOT NULL DEFAULT 0, taux_tva NUMERIC DEFAULT 18, montant_tva NUMERIC GENERATED ALWAYS AS (ROUND(montant_ht*taux_tva/100,0)) STORED, montant_ttc NUMERIC GENERATED ALWAYS AS (ROUND(montant_ht*(1+taux_tva/100),0)) STORED, statut TEXT DEFAULT 'en_attente', mode_paiement TEXT, notes TEXT, agent_id UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE paiements ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), facture_id UUID REFERENCES factures(id), montant NUMERIC NOT NULL, mode TEXT NOT NULL, date_paie DATE DEFAULT CURRENT_DATE, reference TEXT, agent_id UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE journal_comptable ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), date_op DATE NOT NULL DEFAULT CURRENT_DATE, numero_piece TEXT NOT NULL, compte_syscohada TEXT NOT NULL, intitule_compte TEXT NOT NULL, libelle TEXT NOT NULL, debit NUMERIC DEFAULT 0, credit NUMERIC DEFAULT 0, journal TEXT DEFAULT 'VT', agent_id UUID REFERENCES profiles(id), created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE fournisseurs ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), nom TEXT NOT NULL, categorie TEXT, telephone TEXT, email TEXT, solde_du NUMERIC DEFAULT 0, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Storage bucket sécurisé INSERT INTO storage.buckets (id,name,public) VALUES ('documents-clients','documents-clients',false);
5. Sécurité Row Level Security (RLS)
-- Activer RLS ALTER TABLE profiles ENABLE ROW LEVEL SECURITY; ALTER TABLE clients ENABLE ROW LEVEL SECURITY; ALTER TABLE dossiers ENABLE ROW LEVEL SECURITY; ALTER TABLE engagements ENABLE ROW LEVEL SECURITY; ALTER TABLE documents_clients ENABLE ROW LEVEL SECURITY; ALTER TABLE factures ENABLE ROW LEVEL SECURITY; ALTER TABLE journal_comptable ENABLE ROW LEVEL SECURITY; -- Fonction rôle CREATE OR REPLACE FUNCTION get_my_role() RETURNS user_role AS $$ SELECT role FROM profiles WHERE id = auth.uid(); $$ LANGUAGE sql SECURITY DEFINER STABLE; -- Clients : lecture tous, écriture gestionnaire+ CREATE POLICY "clients_read" ON clients FOR SELECT USING (auth.uid() IS NOT NULL); CREATE POLICY "clients_write" ON clients FOR INSERT WITH CHECK ( get_my_role() IN ('administrateur','gestionnaire','employe')); CREATE POLICY "clients_upd" ON clients FOR UPDATE USING ( get_my_role() IN ('administrateur','gestionnaire')); -- Journal : lecture comptable+, écriture admin/comptable CREATE POLICY "journal_read" ON journal_comptable FOR SELECT USING ( get_my_role() IN ('administrateur','gestionnaire','comptable')); CREATE POLICY "journal_write" ON journal_comptable FOR INSERT WITH CHECK ( get_my_role() IN ('administrateur','comptable')); -- Trigger updated_at CREATE OR REPLACE FUNCTION set_updated_at() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER trg_clients BEFORE UPDATE ON clients FOR EACH ROW EXECUTE FUNCTION set_updated_at(); CREATE TRIGGER trg_dossiers BEFORE UPDATE ON dossiers FOR EACH ROW EXECUTE FUNCTION set_updated_at(); -- Storage RLS CREATE POLICY "doc_upload" ON storage.objects FOR INSERT WITH CHECK ( bucket_id='documents-clients' AND auth.uid() IS NOT NULL); CREATE POLICY "doc_read" ON storage.objects FOR SELECT USING ( bucket_id='documents-clients' AND auth.uid() IS NOT NULL);
